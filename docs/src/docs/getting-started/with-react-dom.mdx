---
title: With React DOM
---

If you're using React DOM, you can simply use the Game component within your React app

```jsx live=true noInline=true
import { Game, Scene, Text } from 'react-phaser-fiber'
import { render } from 'react-dom'

function MyGame() {
  return (
    <Game width={400} height={400}>
      <Scene sceneKey="main">
        <Text
          x={140}
          y={150}
          text="Hello World"
          style={{ color: 'white' }}
        />
      </Scene>
    </Game>
  )
}

render(<MyGame />, document.getElementById('root'))
```

# Canvas

You can also customize the canvas that Phaser renders into

```jsx live=true customCanvas=true
import {
  Game,
  Scene,
  Text,
  Canvas,
} from 'react-phaser-fiber'

function MyGame() {
  return (
    <Canvas
      style={{
        border: '5px red solid',
        transform: 'scale(0.5)',
      }}
    >
      <Game width={400} height={400}>
        <Scene sceneKey="main">
          <Text
            x={140}
            y={150}
            text="Hello World"
            style={{ color: 'white' }}
          />
        </Scene>
      </Game>
    </Canvas>
  )
}
```
